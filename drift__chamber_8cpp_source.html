152 
153  // Now we fill the sectors object which holds all these
154  // core parameters. Here, many numbers will be redundant.
155  // It is expected that this will change once efficiency
156  // alignment and other calibrations are taken into effect.
157  _sectors.clear();
158 
159  for (size_t sec=0; sec<nsectors; sec++)
160  {

178  region._thtilt = thtilt[reg] * deg2rad;
179  region._xdist = xdist[reg];
180

182  {
183

186  int nslyr = slyr;

194
